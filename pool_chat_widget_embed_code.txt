<script>
  function toggleChat() {
    const chat = document.getElementById("poolChatWidget");
    chat.style.display = (chat.style.display === "flex") ? "none" : "flex";
  }

  function handleEnter(e) {
    if (e.key === "Enter") {
      e.preventDefault();
      sendPoolMsg();
    }
  }

  async function sendPoolMsg() {
    const input = document.getElementById("userInput");
    const body = document.getElementById("chatBody");
    const msg = input.value.trim();
    if (!msg) return;

    const userBubble = document.createElement("div");
    userBubble.className = "msg user";
    userBubble.textContent = msg;
    body.appendChild(userBubble);

    const botBubble = document.createElement("div");
    botBubble.className = "msg bot";
    botBubble.textContent = "Thinking...";
    body.appendChild(botBubble);

    input.value = "";
    body.scrollTop = body.scrollHeight;

    try {
      const res = await fetch("https://cocktail-pool-chatbot.onrender.com/chat", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ message: msg })
      });

      const data = await res.json();
      botBubble.innerHTML = (data.reply || "Hmm... I didnâ€™t catch that.")
        .replace(/(https?:\/\/\S+)/g, '<a href="$1" target="_blank">$1</a>');
    } catch (err) {
      botBubble.textContent = "Sorry â€” Iâ€™m having trouble reaching the server.";
    }

    body.scrollTop = body.scrollHeight;
  }
</script>
